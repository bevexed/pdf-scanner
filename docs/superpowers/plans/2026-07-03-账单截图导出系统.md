# 账单截图导出系统 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 本地离线工具,导入 FedEx 账单 PDF 建索引,输入运单号判断有无杂费/关税,截图打码后导出,打包成单 Windows exe。

**Architecture:** PyMuPDF 逐页读文字坐标 → 以「空运提单AWB」锚定切每票区域(三列布局:左基础信息/中寄件人/右收件人/费用行金额右对齐)→ 判定并预计算杂费·关税结果存 SQLite → 查询时渲染区域并对寄件人及金额高斯模糊 → Flask 单页界面 → PyInstaller 打包。

**Tech Stack:** Python 3.11, PyMuPDF (fitz), Pillow, Flask, sqlite3(内置), pytest, PyInstaller

**真实数据坐标基准**(测试用,来自 `var/CN_XXXXX0612.0002.08051642551452.pdf`):
- 页尺寸 595×842;第 56 页每票高约 125pt,AWB 标签 x≈42、单号在 x≈128
- 寄件人 x≈243~395,收件人 x≈408+,费用标签 x≈300,金额 `￥` 在 x≈533~564
- 第 56 页 `889635416339` 有杂费(超范围派送费 A);第 102 页 `889915287001` 有关税(Customs);第 56 页 `889414201705` 无杂费

```python
# core/classifier.py
from core import config

def _is_fee_row(line: str) -> bool:
    """含金额(￥) 的行视为费用行。合计/Total 行排除(不是费用科目)。"""
    if not config.AMOUNT_RE.search(line):
        return False
    if "合计" in line or "Total" in line:
        return False
    return True

def _is_normal(line: str) -> bool:
    low = line.lower()
    return any(k.lower() in low for k in config.NORMAL_FEES)

def has_extra_fee(region_text: str) -> bool:
    """费用行里出现非白名单科目 → 有杂费。"""
    for raw in region_text.splitlines():
        ln = raw.strip()
        if _is_fee_row(ln) and not _is_normal(ln):
            return True
    return False

def has_duty(region_text: str) -> bool:
    return any(k in region_text for k in config.DUTY_KEYWORDS)
```

> 注:金额与科目名可能不在同一行(PDF 文本流里标签与￥分行)。此处判定用「一票内是否存在既非白名单、又是费用科目的行」。由于科目名单独成行且不含￥,需改为「科目名行 + 下一行是￥」的判定 —— 见 Step 3b 修正。

- [ ] **Step 3b: 修正为「科目名行 + 紧邻金额行」判定**

真实文本里科目名和金额分两行(如 `超范围派送费 A ...` / `￥23.00`)。改为逐行:若某行**下一行**是纯金额行,则该行是费用科目行,再判断是否白名单。

```python
# core/classifier.py  —— 替换 has_extra_fee
def has_extra_fee(region_text: str) -> bool:
    lines = [ln.strip() for ln in region_text.splitlines()]
    for i, ln in enumerate(lines):
        nxt = lines[i + 1] if i + 1 < len(lines) else ""
        # 该行是科目名(非空、非金额、非合计),下一行是纯金额
        if not ln or config.AMOUNT_RE.fullmatch(nxt) is None:
            continue
        if "合计" in ln or "Total" in ln:
            continue
        if config.AMOUNT_RE.search(ln):  # 本行已含金额,跳过(避免误判)
            continue
        if not _is_normal(ln):
            return True
    return False
```

```python
# core/pdf_parser.py
import fitz
from core import config
from core.classifier import has_extra_fee, has_duty

# 金额科目标签 → mask kind(用于打码金额值)
_AMOUNT_LABELS = [
    ("运费", "freight"), ("折扣", "discount"),
    ("燃油附加费", "fuel"), ("合计", "total"),
]

def _words_in(words, y0, y1):
    return [w for w in words if y0 <= w[1] < y1]

def _row_amount_rect(words, label_y):
    """同一行(y接近 label_y)、x>AMOUNT_MIN_X 的 ￥ word 的坐标。"""
    for w in words:
        if abs(w[1] - label_y) <= config.ROW_Y_TOL and w[0] >= config.AMOUNT_MIN_X \
                and "￥" in w[4]:
            return [w[0], w[1], w[2], w[3]]
    return None

def _sender_rect(words, top_y, bottom_y):
    """寄件人列包络框:x∈SENDER_X_RANGE、y∈[top_y,bottom_y) 的 word 并集。"""
    xs0, xs1 = config.SENDER_X_RANGE
    sel = [w for w in words if top_y <= w[1] < bottom_y
           and w[0] >= xs0 and w[2] <= xs1 + 5]
    if not sel:
        return None
    return [min(w[0] for w in sel), min(w[1] for w in sel),
            max(w[2] for w in sel), max(w[3] for w in sel)]

def _parse_page(page, page_no):
    words = page.get_text("words")  # (x0,y0,x1,y1,text,block,line,word_no)
    # 1) 锚点:找所有 "AWB" 标签行,同 y 找 12 位单号
    awb_anchors = []
    for w in words:
        if w[4] == "AWB":
            num = next((v for v in words
                        if abs(v[1] - w[1]) <= config.ROW_Y_TOL
                        and config.AWB_RE.match(v[4])), None)
            if num:
                awb_anchors.append((w[1], num[4]))  # (y0, awb)
    awb_anchors.sort(key=lambda t: t[0])

    results = []
    for i, (y_top, awb) in enumerate(awb_anchors):
        y_bottom = awb_anchors[i + 1][0] if i + 1 < len(awb_anchors) else page.rect.y1
        rect = fitz.Rect(page.rect.x0, y_top - config.TICKET_TOP_PAD,
                         page.rect.x1, y_bottom - config.TICKET_BOTTOM_PAD)
        tw = _words_in(words, rect.y0, rect.y1)
        region_text = page.get_textbox(rect)

        # 打码矩形:寄件人 + 各金额
        mask_rects = []
        # 寄件人区:从本票顶部到"运费"标签行之前
        freight_y = next((w[1] for w in tw if w[4] == "运费"), rect.y1)
        s = _sender_rect(tw, rect.y0, freight_y)
        if s:
            mask_rects.append({"kind": "sender", "rect": s})
        for label, kind in _AMOUNT_LABELS:
            ly = next((w[1] for w in tw if w[4] == label), None)
            if ly is None:
                continue
            amt = _row_amount_rect(tw, ly)
            if amt:
                mask_rects.append({"kind": kind, "rect": amt})

        results.append({
            "awb": awb,
            "page_no": page_no,
            "rect": [rect.x0, rect.y0, rect.x1, rect.y1],
            "region_text": region_text,
            "mask_rects": mask_rects,
            "has_extra": has_extra_fee(region_text),
            "has_duty": has_duty(region_text),
        })
    return results

def parse_pdf(pdf_path, page_range=None):
    """返回该 PDF 所有票。page_range=(start,end) 半开区间,用于测试/进度分批。
    进度回调:可选 progress_cb(done_pages, total_pages)。"""
    doc = fitz.open(pdf_path)
    lo = 0 if page_range is None else page_range[0]
    hi = len(doc) if page_range is None else min(page_range[1], len(doc))
    out = []
    for pno in range(lo, hi):
        out.extend(_parse_page(doc[pno], pno))
    doc.close()
    return out
```

```python
# core/indexer.py
import sqlite3, json, datetime

_SCHEMA = """
CREATE TABLE IF NOT EXISTS billing_index (
  id INTEGER PRIMARY KEY,
  awb TEXT, pdf_file TEXT, page_no INTEGER,
  rect TEXT, region_text TEXT, mask_rects TEXT,
  bill_type TEXT, has_extra INTEGER, has_duty INTEGER
);
CREATE INDEX IF NOT EXISTS idx_awb ON billing_index(awb);
CREATE TABLE IF NOT EXISTS imported_files (
  pdf_file TEXT PRIMARY KEY, imported_at TEXT, page_count INTEGER, ticket_count INTEGER
);
"""

def _conn(db_path):
    c = sqlite3.connect(db_path)
    c.row_factory = sqlite3.Row
    return c

def init_db(db_path):
    with _conn(db_path) as c:
        c.executescript(_SCHEMA)

def insert_tickets(db_path, pdf_file, tickets):
    with _conn(db_path) as c:
        c.executemany(
            """INSERT INTO billing_index
               (awb,pdf_file,page_no,rect,region_text,mask_rects,bill_type,has_extra,has_duty)
               VALUES (?,?,?,?,?,?,?,?,?)""",
            [(t["awb"], pdf_file, t["page_no"], json.dumps(t["rect"]),
              t["region_text"], json.dumps(t["mask_rects"]), "freight",
              int(t["has_extra"]), int(t["has_duty"])) for t in tickets])

def lookup(db_path, awb):
    with _conn(db_path) as c:
        r = c.execute("SELECT * FROM billing_index WHERE awb=? LIMIT 1", (awb,)).fetchone()
    if not r:
        return None
    return {"awb": r["awb"], "pdf_file": r["pdf_file"], "page_no": r["page_no"],
            "rect": json.loads(r["rect"]), "mask_rects": json.loads(r["mask_rects"]),
            "has_extra": bool(r["has_extra"]), "has_duty": bool(r["has_duty"])}

def is_imported(db_path, pdf_file, page_count):
    with _conn(db_path) as c:
        r = c.execute("SELECT 1 FROM imported_files WHERE pdf_file=? AND page_count=?",
                      (pdf_file, page_count)).fetchone()
    return r is not None

def mark_imported(db_path, pdf_file, page_count, ticket_count):
    with _conn(db_path) as c:
        c.execute("INSERT OR REPLACE INTO imported_files VALUES (?,?,?,?)",
                  (pdf_file, datetime.datetime.now().isoformat(), page_count, ticket_count))

def total_tickets(db_path):
    with _conn(db_path) as c:
```

```python
# core/exporter.py
import fitz, os
from PIL import Image, ImageFilter
from core import config

def export_masked(pdf_path, page_no, rect, mask_rects, out_path):
    """渲染区域→对 mask_rects 高斯模糊→存 PNG。mask_rects: [{kind,rect:[x0,y0,x1,y1]}]。"""
    z = config.RENDER_ZOOM
    doc = fitz.open(pdf_path)
    page = doc[page_no]
    clip = fitz.Rect(*rect)
    pix = page.get_pixmap(matrix=fitz.Matrix(z, z), clip=clip)
    img = Image.frombytes("RGB", (pix.width, pix.height), pix.samples)
    for m in mask_rects:
        r = m["rect"]
        mx0 = int((r[0] - rect[0]) * z); my0 = int((r[1] - rect[1]) * z)
        mx1 = int((r[2] - rect[0]) * z); my1 = int((r[3] - rect[1]) * z)
        box = (max(0, mx0), max(0, my0), min(img.width, mx1), min(img.height, my1))
        if box[2] > box[0] and box[3] > box[1]:
            region = img.crop(box).filter(ImageFilter.GaussianBlur(config.BLUR_RADIUS))
            img.paste(region, box)
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    img.save(out_path)
    doc.close()
    return out_path
```

```python
# app.py
import os, threading, webbrowser, zipfile, io
from flask import (Flask, request, jsonify, render_template,
                   send_from_directory, send_file)
from core import config, indexer, exporter
from core.pdf_parser import parse_pdf
import fitz

app = Flask(__name__, template_folder="web/templates", static_folder="web/static")

# 全局进度状态(单用户本地工具,简单够用)
_progress = {"state": "idle", "done": 0, "total": 0, "tickets": 0, "message": ""}

def _ensure_dirs():
    for d in (config.DATA_DIR, config.PDF_DIR, config.EXPORT_DIR):
        os.makedirs(d, exist_ok=True)
    indexer.init_db(config.DB_PATH)

def _pdf_path(pdf_file):
    return os.path.join(config.PDF_DIR, pdf_file)

@app.route("/")
def index():
    return render_template("index.html", total=indexer.total_tickets(config.DB_PATH))

@app.route("/progress")
def progress():
    return jsonify(_progress)

def _do_import(save_path, pdf_file):
    global _progress
    try:
        doc = fitz.open(save_path); total = len(doc); doc.close()
        if indexer.is_imported(config.DB_PATH, pdf_file, total):
            _progress = {"state": "done", "done": total, "total": total,
                         "tickets": indexer.total_tickets(config.DB_PATH),
                         "message": "该文件已导入,跳过"}
            return
        _progress = {"state": "parsing", "done": 0, "total": total, "tickets": 0, "message": ""}
        STEP = 200
        for lo in range(0, total, STEP):
            tickets = parse_pdf(save_path, (lo, min(lo + STEP, total)))
            indexer.insert_tickets(config.DB_PATH, pdf_file, tickets)
            _progress["done"] = min(lo + STEP, total)
        indexer.mark_imported(config.DB_PATH, pdf_file, total,
                              indexer.total_tickets(config.DB_PATH))
        _progress["state"] = "done"
        _progress["tickets"] = indexer.total_tickets(config.DB_PATH)
    except Exception as e:
        _progress["state"] = "error"; _progress["message"] = str(e)

@app.route("/import", methods=["POST"])
def import_pdf():
    f = request.files["file"]
    pdf_file = os.path.basename(f.filename)
    save_path = _pdf_path(pdf_file)
    f.save(save_path)
    threading.Thread(target=_do_import, args=(save_path, pdf_file), daemon=True).start()
    return jsonify({"ok": True})

@app.route("/query", methods=["POST"])
def query():
    body = request.get_json()
    awbs = [a.strip() for a in body["awbs"] if a.strip()]
    mode = body.get("mode", "fee")           # 'fee' 杂费 / 'duty' 关税
    results = []
    for awb in awbs:
        rec = indexer.lookup(config.DB_PATH, awb)
        if not rec:
            results.append({"awb": awb, "status": "未找到"}); continue
        hit = rec["has_extra"] if mode == "fee" else rec["has_duty"]
        if not hit:
            results.append({"awb": awb, "status": "无杂费" if mode == "fee" else "无关税"})
            continue
        masks = rec["mask_rects"]
        if mode == "duty":
            masks = [m for m in masks if m["kind"] == "sender"]
        out = os.path.join(config.EXPORT_DIR, f"{awb}.png")
        exporter.export_masked(_pdf_path(rec["pdf_file"]), rec["page_no"],
                               rec["rect"], masks, out)
        results.append({"awb": awb, "status": "有杂费" if mode == "fee" else "有关税",
                        "image": f"/exports/{awb}.png"})
    return jsonify(results)

@app.route("/exports/<path:name>")
def exports(name):
    return send_from_directory(config.EXPORT_DIR, name)

@app.route("/export_zip", methods=["POST"])
def export_zip():
    awbs = request.get_json()["awbs"]
    buf = io.BytesIO()
    with zipfile.ZipFile(buf, "w") as z:
        for awb in awbs:
            p = os.path.join(config.EXPORT_DIR, f"{awb}.png")
            if os.path.exists(p):
                z.write(p, f"{awb}.png")
    buf.seek(0)
    return send_file(buf, mimetype="application/zip",
                     as_attachment=True, download_name="exports.zip")

def open_browser():
    webbrowser.open("http://127.0.0.1:5000")

if __name__ == "__main__":
    _ensure_dirs()
    threading.Timer(1.0, open_browser).start()
    app.run(host="127.0.0.1", port=5000)
```

```javascript
let lastHits = [];

async function doImport() {
  const f = document.getElementById('pdf').files[0];
  if (!f) return alert('请选择 PDF');
  const fd = new FormData(); fd.append('file', f);
  await fetch('/import', {method: 'POST', body: fd});
  pollProgress();
}

async function pollProgress() {
  const p = document.getElementById('progress');
  const timer = setInterval(async () => {
    const s = await (await fetch('/progress')).json();
    if (s.state === 'parsing') p.textContent = `解析中 ${s.done}/${s.total} 页…`;
    else if (s.state === 'done') { p.innerHTML = `已入库 <b>${s.tickets}</b> 票 ${s.message||''}`; clearInterval(timer); }
    else if (s.state === 'error') { p.textContent = '出错: ' + s.message; clearInterval(timer); }
  }, 800);
}

async function doQuery() {
  const mode = document.querySelector('input[name=mode]:checked').value;
  const awbs = document.getElementById('awbs').value.split('\n').map(x=>x.trim()).filter(Boolean);
  const res = await (await fetch('/query', {method:'POST',
    headers:{'Content-Type':'application/json'}, body: JSON.stringify({awbs, mode})})).json();
  const tb = document.querySelector('#table tbody'); tb.innerHTML = '';
  lastHits = [];
  for (const r of res) {
    const tr = document.createElement('tr');
    let op = '';
    if (r.image) { op = `<a href="${r.image}" target="_blank">预览</a>
      <a href="${r.image}" download="${r.awb}.png">导出</a>`; lastHits.push(r.awb); }
    tr.innerHTML = `<td>${r.awb}</td><td>${r.status}</td><td>${op}</td>`;
    tb.appendChild(tr);
  }
}

async function exportZip() {
  if (!lastHits.length) return alert('无可导出结果');
  const resp = await fetch('/export_zip', {method:'POST',
    headers:{'Content-Type':'application/json'}, body: JSON.stringify({awbs: lastHits})});
  const blob = await resp.blob();
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob); a.download = 'exports.zip'; a.click();
}
```
